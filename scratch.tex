Ex 1 1.a)
\begin{align*}
  5 + 1 * 3 & = 5 + (1 * 3) \\
            & = 5 + 3       \\
            & = 8
\end{align*}

Ex 1 1.b)
\begin{align*}
  11 - 2 * 5 & = 11 - (2 * 5) \\
             & = 11 - 10      \\
             & = 1
\end{align*}

Ex 1 1.c)
\begin{align*}
  18 - 10 \div 5 & = 18 - (10 \div  5) \\
                 & = 18 - 2            \\
                 & = 16
\end{align*}

Ex 1 1.n)
\begin{align*}
  5 * 6 - 8 \div 2 & = (5 * 6) - (8 \div 2) \\
                   & = 30 - 4               \\
                   & = 26
\end{align*}

Ex 1 1.s)
\begin{align*}
  48 \div 3 - 7 * 2 & = (48 \div 3) - (7 * 2) \\
                    & = 16 - 14               \\
                    & = 2
\end{align*}

Ex 1 2.h)
\begin{align*}
  \frac{36 \div (11 - 2)}{8 - 8 \div 2} & = \frac{36 \div (11 - 2)}{8 - (8 \div 2)} \\
                                        & = \frac{36 \div 9}{8 - 4}                 \\
                                        & = \frac{4}{4}                             \\
                                        & = 1
\end{align*}
